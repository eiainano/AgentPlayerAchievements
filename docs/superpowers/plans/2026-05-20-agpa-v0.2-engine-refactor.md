```typescript
import type { AchievementDefinition, Condition, ConditionType } from './types.js';

function parseYAML(text: string): { definitions: AchievementDefinition[] } {
  const lines = text.split('\n');
  const root: { definitions: AchievementDefinition[] } = { definitions: [] };
  let current: AchievementDefinition | null = null;
  let currentCond: Partial<Condition> | null = null;
  let inConditions = false;

  for (const line of lines) {
    if (/^\s*#/.test(line) || /^\s*$/.test(line)) continue;

    const indent = line.match(/^(\s*)/)![1].length;

    if (indent === 0 && line.trim().startsWith('definitions:')) continue;

    if (indent === 2 && line.trim().startsWith('- id:')) {
      if (current) root.definitions.push(current);
      current = {
        id: '',
        name: '',
        description: '',
        icon: '🏆',
        category: 'other',
        rarity: 'common',
        conditions: [],
      };
      current.id = line.match(/- id:\s*(.+)/)![1].trim();
      inConditions = false;
      continue;
    }

    if (!current) continue;

    if (indent === 4 && !inConditions) {
      const m = line.match(/^\s+(\w+):\s*(.*)/);
      if (m) {
        const [, key, val] = m;
        const trimmed = val.trim();
        if (key === 'hidden') current.hidden = trimmed === 'true';
        else if (key === 'progress_trackable') current.progress_trackable = trimmed === 'true';
        else if (key === 'conditions') {
          inConditions = true;
          current.conditions = [];
        } else if (key === 'icon' || key === 'name' || key === 'name_cn' ||
                   key === 'description' || key === 'description_cn' ||
                   key === 'category' || key === 'rarity' || key === 'set_id') {
          (current as Record<string, unknown>)[key] = trimmed.replace(/^["']|["']$/g, '');
        }
      }
      continue;
    }

    if (inConditions) {
      if (indent === 6 && line.trim().startsWith('- type:')) {
        currentCond = { value: 0 };
        current.conditions.push(currentCond as Condition);
        currentCond.type = line.match(/type:\s*(.+)/)![1].trim() as ConditionType;
        continue;
      }
      if (currentCond && indent === 8) {
        const m = line.match(/^\s+(\w+):\s*(.*)/);
        if (m) {
          const [, key, val] = m;
          let trimmed: unknown = val.trim().replace(/^["']|["']$/g, '');
          if (key === 'value' && !isNaN(Number(trimmed))) trimmed = Number(trimmed);
          (currentCond as Record<string, unknown>)[key] = trimmed;
        }
        continue;
      }
      if (indent <= 4 && currentCond) {
        inConditions = false;
        currentCond = null;
      }
    }
  }

  if (current) root.definitions.push(current);
  return root;
}
```

```typescript
import type { TrackedEvent, Condition, ConditionOperator, EvaluationResult } from './types.js';

function parseWindow(w: string): number {
  const m = /(\d+)\s*(h|d|m)/.exec(w);
  if (!m) return 86400000;
  const n = Number(m[1]);
  switch (m[2]) {
    case 'h': return n * 3600000;
    case 'd': return n * 86400000;
    case 'm': return n * 60000;
    default: return 86400000;
  }
}

function matchFilter(event: TrackedEvent, filter: string): boolean {
  try {
    const fn = new Function('e', `with(e){ return ${filter}; }`);
    const ctx = {
      event_type: event.event_type,
      tool_name: event.payload?.tool_name || '',
      file_type: event.payload?.file_type || '',
      command: event.payload?.command || '',
      file_path: event.payload?.file_path || '',
      agent_involved: event.payload?.agent_involved || false,
      manual_edits: event.payload?.manual_edits || 0,
      issues_found: event.payload?.issues_found || 0,
    };
    return fn(ctx);
  } catch {
    return true;
  }
}

function getField(event: TrackedEvent, field: string): string {
  return String(event.payload?.[field] || '');
}

// ── Evaluators ────────────────────────────────────────────────────

function evalCounter(events: TrackedEvent[], cond: Condition): EvaluationResult {
  let count = 0;
  for (const e of events) {
    if (e.event_type !== cond.event) continue;
    if (cond.filter && !matchFilter(e, cond.filter)) continue;
    count++;
  }
  const target = cond.value;
  const op: ConditionOperator = cond.operator || '>=';
  const met = op === '>=' ? count >= target : op === '==' ? count === target : op === '>' ? count > target : count < target;
  return { met, progress: count, target };
}

function evalThreshold(events: TrackedEvent[], cond: Condition): EvaluationResult {
  const windowMs = parseWindow(cond.window || '24h');
  const now = Date.now();
  let count = 0;
  for (const e of events) {
    if (e.event_type !== cond.event) continue;
    if (now - new Date(e.timestamp).getTime() > windowMs) continue;
    count++;
  }
  const target = cond.value;
  return { met: count >= target, progress: count, target };
}

function evalStreak(events: TrackedEvent[], cond: Condition): EvaluationResult {
  const days = new Set<string>();
  for (const e of events) {
    if (e.event_type !== cond.event) continue;
    days.add(e.timestamp.slice(0, 10));
  }
  const sorted = [...days].sort().reverse();
  if (sorted.length === 0) return { met: false, progress: 0, target: cond.value };

  let streak = 1;
  for (let i = 0; i < sorted.length - 1; i++) {
    const d1 = new Date(sorted[i]!);
    const d2 = new Date(sorted[i + 1]!);
    if ((d1.getTime() - d2.getTime()) / 86400000 <= 1) streak++;
    else break;
  }
  const target = cond.value;
  return { met: streak >= target, progress: streak, target };
}

function evalSequence(events: TrackedEvent[], cond: Condition): EvaluationResult {
  const seq = cond.sequence || [];
  if (seq.length === 0) return { met: false, progress: 0, target: seq.length };
  let si = 0;
  for (const e of events) {
    if (si >= seq.length) break;
    if (e.event_type === seq[si]) si++;
  }
  return { met: si >= seq.length, progress: si, target: seq.length };
}

function evalDistinctCount(events: TrackedEvent[], cond: Condition): EvaluationResult {
  const values = new Set<string>();
  for (const e of events) {
    if (e.event_type !== cond.event) continue;
    const val = cond.field ? getField(e, cond.field) : null;
    if (val) values.add(val);
  }
  const target = cond.value;
  return { met: values.size >= target, progress: values.size, target };
}

function evalEvent(events: TrackedEvent[], cond: Condition): EvaluationResult {
  for (const e of events) {
    if (e.event_type === cond.event) return { met: true, progress: 1, target: 1 };
  }
  return { met: false, progress: 0, target: 1 };
}

// ── Dispatcher ────────────────────────────────────────────────────

export function evaluateCondition(cond: Condition, events: TrackedEvent[]): EvaluationResult {
  switch (cond.type) {
    case 'counter': return evalCounter(events, cond);
    case 'threshold': return evalThreshold(events, cond);
    case 'streak': return evalStreak(events, cond);
    case 'sequence': return evalSequence(events, cond);
    case 'distinct_count': return evalDistinctCount(events, cond);
    case 'event': return evalEvent(events, cond);
    default: return { met: false, progress: 0, target: 0 };
  }
}

export function evaluateAll(definitions: Array<{ id: string; conditions: Condition[] }>, events: TrackedEvent[], unlocked: Record<string, string>): string[] {
  const newlyUnlocked: string[] = [];
  for (const def of definitions) {
    if (unlocked[def.id]) continue;
    const allMet = def.conditions.every(c => evaluateCondition(c, events).met);
    if (allMet) newlyUnlocked.push(def.id);
  }
  return newlyUnlocked;
}
```

Note: The `evaluateAll` function replaces the condition-checking loop from `engine.poll()` — it takes `definitions`, `events`, and `unlocked` as pure parameters and returns IDs of newly unlocked achievements.

### Task 4: Create `src/engine/counter-cache.ts`

**File:** Create `src/engine/counter-cache.ts`

Implement the incremental counter cache. On `track()`, increment counts in O(1). On `poll()`, read counts in O(1). `rebuild()` does a full pass when cache is missing.

- [ ] **Step 1: Write `src/engine/counter-cache.ts`**

```typescript
import type { TrackedEvent } from './types.js';
import type { SerializedCache } from './types.js';

function hashFilter(filter?: string): string {
  if (!filter) return '';
  let hash = 0;
  for (let i = 0; i < filter.length; i++) {
    hash = ((hash << 5) - hash + filter.charCodeAt(i)) | 0;
  }
  return String(hash);
}

export class CounterCache {
  eventCounts: Record<string, number> = {};
  filterCounts: Record<string, number> = {};
  streaks: Record<string, string[]> = {};
  distinctValues: Record<string, Set<string>> = {};
  private initialized = false;

  constructor(snapshot?: SerializedCache) {
    if (snapshot) this.deserialize(snapshot);
    this.initialized = !!snapshot;
  }

  /** Call on every track() — O(1) */
  onTrack(event: TrackedEvent): void {
    // event count
    this.eventCounts[event.event_type] = (this.eventCounts[event.event_type] || 0) + 1;

    // streak tracking: track dates by event type
    const day = event.timestamp.slice(0, 10);
    if (!this.streaks[event.event_type]) this.streaks[event.event_type] = [];
    const days = this.streaks[event.event_type]!;
    if (!days.includes(day)) days.push(day);

    // distinct values for payload fields
    if (event.payload) {
      for (const [key, val] of Object.entries(event.payload)) {
        if (typeof val === 'string') {
          const cacheKey = `${event.event_type}:${key}`;
          if (!this.distinctValues[cacheKey]) this.distinctValues[cacheKey] = new Set();
          this.distinctValues[cacheKey]!.add(val);
        }
      }
    }
  }

  /** Get count of events by type */
  getCount(eventType: string): number {
    return this.eventCounts[eventType] || 0;
  }

  /** Get streak length for streak condition */
  getStreakLength(eventType: string): number {
    const days = this.streaks[eventType];
    if (!days || days.length === 0) return 0;

    const sorted = [...days].sort().reverse();
    let streak = 1;
    for (let i = 0; i < sorted.length - 1; i++) {
      const d1 = new Date(sorted[i]!);
      const d2 = new Date(sorted[i + 1]!);
      if ((d1.getTime() - d2.getTime()) / 86400000 <= 1) streak++;
      else break;
    }
    return streak;
  }

  /** Get distinct value count for a field on an event type */
  getDistinctCount(eventType: string, field: string): number {
    const key = `${eventType}:${field}`;
    return this.distinctValues[key]?.size || 0;
  }

  /** Serialize to JSON for persistence */
  serialize(): SerializedCache {
    const dvs: Record<string, string[]> = {};
    for (const [key, set] of Object.entries(this.distinctValues)) {
      dvs[key] = [...set];
    }
    return {
      eventCounts: { ...this.eventCounts },
      filterCounts: { ...this.filterCounts },
      streaks: Object.fromEntries(
        Object.entries(this.streaks).map(([k, v]) => [k, [...v]])
      ),
      distinctValues: dvs,
    };
  }

  /** Deserialize from persisted state */
  private deserialize(snapshot: SerializedCache): void {
    this.eventCounts = { ...snapshot.eventCounts };
    this.filterCounts = { ...snapshot.filterCounts };
    this.streaks = Object.fromEntries(
      Object.entries(snapshot.streaks || {}).map(([k, v]) => [k, [...v]])
    );
    const dvs: Record<string, Set<string>> = {};
    for (const [key, arr] of Object.entries(snapshot.distinctValues || {})) {
      dvs[key] = new Set(arr);
    }
    this.distinctValues = dvs;
  }

  /** Full rebuild from event log (O(n), one-time on init when cache is missing) */
  rebuild(events: TrackedEvent[]): void {
    this.eventCounts = {};
    this.filterCounts = {};
    this.streaks = {};
    this.distinctValues = {};
    for (const event of events) {
      this.onTrack(event);
    }
  }
}
```

```typescript
import * as fs from 'fs';
import * as path from 'path';
import type { AchievementState, TrackedEvent, SerializedCache } from './types.js';

export class Store {
  readonly stateDir: string;

  constructor(stateDir: string) {
    this.stateDir = stateDir;
  }

  ensureDir(): void {
    fs.mkdirSync(this.stateDir, { recursive: true });
  }

  /** Load state, events, and cache from disk */
  load(): { state: AchievementState; events: TrackedEvent[]; cache: SerializedCache | null } {
    this.ensureDir();

    // Load state
    const statePath = path.join(this.stateDir, 'state.json');
    let state: AchievementState;
    try {
      if (fs.existsSync(statePath)) {
        state = JSON.parse(fs.readFileSync(statePath, 'utf-8'));
      } else {
        state = { unlocked: {}, stats: { total_unlocked: 0 } };
      }
    } catch {
      state = { unlocked: {}, stats: { total_unlocked: 0 } };
    }

    // Load event log
    const logPath = path.join(this.stateDir, 'event.log');
    let events: TrackedEvent[] = [];
    try {
      if (fs.existsSync(logPath)) {
        const lines = fs.readFileSync(logPath, 'utf-8').trim().split('\n');
        events = lines.filter(Boolean).map(line => {
          try { return JSON.parse(line) as TrackedEvent; } catch { return null; }
        }).filter((e): e is TrackedEvent => e !== null);
      }
    } catch {
      events = [];
    }

    // Load cache
    const cachePath = path.join(this.stateDir, 'cache.json');
    let cache: SerializedCache | null = null;
    try {
      if (fs.existsSync(cachePath)) {
        cache = JSON.parse(fs.readFileSync(cachePath, 'utf-8')) as SerializedCache;
      }
    } catch {
      cache = null;
    }

    return { state, events, cache };
  }

  /** Atomic write: tmp file then rename */
  saveState(state: AchievementState): void {
    this.ensureDir();
    const target = path.join(this.stateDir, 'state.json');
    const tmp = target + '.tmp';
    fs.writeFileSync(tmp, JSON.stringify(state));
    fs.renameSync(tmp, target);
  }

  /** Save cache */
  saveCache(cache: SerializedCache): void {
    this.ensureDir();
    const target = path.join(this.stateDir, 'cache.json');
    const tmp = target + '.tmp';
    fs.writeFileSync(tmp, JSON.stringify(cache));
    fs.renameSync(tmp, target);
  }

  /** Append event to log */
  appendEvent(event: TrackedEvent): void {
    this.ensureDir();
    const logPath = path.join(this.stateDir, 'event.log');
    fs.appendFileSync(logPath, JSON.stringify(event) + '\n');
  }

  /** Full reset */
  reset(): void {
    const statePath = path.join(this.stateDir, 'state.json');
    const logPath = path.join(this.stateDir, 'event.log');
    const cachePath = path.join(this.stateDir, 'cache.json');
    try { if (fs.existsSync(statePath)) fs.unlinkSync(statePath); } catch {}
    try { if (fs.existsSync(logPath)) fs.unlinkSync(logPath); } catch {}
    try { if (fs.existsSync(cachePath)) fs.unlinkSync(cachePath); } catch {}
    this.ensureDir();
  }
}
```

```typescript
  readonly defsPath: string;
  readonly enabledCategories?: string[];

  definitions: AchievementDefinition[] = [];
  events: TrackedEvent[] = [];
  state: AchievementState = { unlocked: {}, stats: { total_unlocked: 0 } };
  unlockedThisPoll: AchievementDefinition[] = [];

  private store: Store;
  private counterCache: CounterCache;

  constructor(opts: EngineOptions = {}) {
    const defaultStateDir = path.join(process.env.HOME || '~', '.agent-achievements');
    this.stateDir = opts.stateDir || defaultStateDir;
    this.defsPath = opts.defsPath || path.join(ROOT, '04-成就定义清单.yaml');
    this.enabledCategories = opts.enabledCategories;
    this.store = new Store(this.stateDir);
    this.counterCache = new CounterCache();
  }

  init(): this {
    // Load YAML definitions
    const yaml = fs.readFileSync(this.defsPath, 'utf-8');
    const parsed = parseYAML(yaml);
    if (parsed.definitions.length === 0) throw new Error('No achievement definitions loaded');

    // Filter by enabled categories if set
    this.definitions = this.enabledCategories
      ? parsed.definitions.filter(d => this.enabledCategories!.includes(d.category))
      : parsed.definitions;

    // Load state, events, cache from store
    const { state, events, cache } = this.store.load();
    this.state = state;
    this.events = events;

    // Initialize counter cache
    this.counterCache = new CounterCache(cache ?? undefined);
    if (!cache) {
      this.counterCache.rebuild(this.events);
      this.store.saveCache(this.counterCache.serialize());
    }

    return this;
  }

  track(eventType: EventType, payload: EventPayload = {}): TrackedEvent {
    const event: TrackedEvent = {
      protocol_version: '1.0',
      event_id: crypto.randomUUID
        ? crypto.randomUUID()
        : `${Date.now()}-${Math.random().toString(36).slice(2)}`,
      timestamp: new Date().toISOString(),
      tool_source: 'claude-code',
      event_type: eventType,
      payload,
      context: { session_id: 'demo-session', model: 'sonnet' },
    };

    this.events.push(event);
    this.store.appendEvent(event);
    this.counterCache.onTrack(event);

    return event;
  }

  poll(): AchievementDefinition[] {
    const unlockedIds = evaluateAll(this.definitions, this.events, this.state.unlocked);
    const newlyUnlocked: AchievementDefinition[] = [];

    for (const id of unlockedIds) {
      const def = this.definitions.find(d => d.id === id);
      if (!def) continue;

      const now = new Date().toISOString();
      this.state.unlocked[id] = now;
      this.state.stats.total_unlocked = (this.state.stats.total_unlocked || 0) + 1;

      newlyUnlocked.push({ ...def, unlocked_at: now, progress: null });
    }

    if (newlyUnlocked.length > 0) {
      this.store.saveState(this.state);
    }

    this.unlockedThisPoll = newlyUnlocked;
    return newlyUnlocked;
  }

  stats(): AchievementStats {
    const total = this.definitions.length;
    const unlocked = Object.keys(this.state.unlocked).length;
    const byCategory: Record<string, { total: number; unlocked: number }> = {};
    const byRarity: Record<string, { total: number; unlocked: number }> = {};

    for (const def of this.definitions) {
      const cat = def.category || 'unknown';
      const rar = def.rarity || 'common';
      if (!byCategory[cat]) byCategory[cat] = { total: 0, unlocked: 0 };
      if (!byRarity[rar]) byRarity[rar] = { total: 0, unlocked: 0 };
      byCategory[cat]!.total++;
      byRarity[rar]!.total++;
      if (this.state.unlocked[def.id]) {
        byCategory[cat]!.unlocked++;
        byRarity[rar]!.unlocked++;
      }
    }

    return {
      total_achievements: total,
      unlocked,
      completion_pct: Math.round((unlocked / total) * 100),
      total_events: this.events.length,
      by_category: byCategory,
      by_rarity: byRarity,
      state_dir: this.stateDir,
    };
  }

  resetState(): void {
    this.state = { unlocked: {}, stats: { total_unlocked: 0 } };
    this.events = [];
    this.counterCache = new CounterCache();
    this.store.reset();
  }
}
```

```typescript
  try {
    if (fs.existsSync(CONFIG_PATH)) {
      const raw = JSON.parse(fs.readFileSync(CONFIG_PATH, 'utf-8'));

    cfg.lang = process.env.AGPA_LANG;
  }

    cfg.enabledCategories = process.env.AGPA_ENABLED_CATEGORIES.split(',').map(s => s.trim()).filter(Boolean);
  }
  if (process.env.AGPA_DEBUG === 'true') {
    cfg.debug = true;
  }

  return cfg;
}

export function saveConfig(partial: Partial<AppConfig>): AppConfig {
  const current = loadConfig();
  const merged = { ...current, ...partial };
  fs.mkdirSync(CONFIG_DIR, { recursive: true });
  fs.writeFileSync(CONFIG_PATH, JSON.stringify(merged, null, 2));
  return merged;
}

export function getConfig(key: keyof AppConfig): string {
```

```typescript
import * as fs from 'fs';

import { z } from 'zod';
import type { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';

function loadPending(stateDir: string): unknown[] {
  const pendingPath = `${stateDir}/pending.json`;
  try {
    if (fs.existsSync(pendingPath)) {
      return JSON.parse(fs.readFileSync(pendingPath, 'utf-8'));
    }
  } catch { /* ignore */ }
  return [];
}

function savePending(stateDir: string, pending: unknown[]): void {
  const pendingPath = `${stateDir}/pending.json`;
  fs.mkdirSync(stateDir, { recursive: true });
  fs.writeFileSync(pendingPath, JSON.stringify(pending, null, 2));
}

function sendMacNotification(title: string, subtitle: string): void {
  const script = `display notification "${subtitle.replace(/"/g, '\\"')}" with title "${title.replace(/"/g, '\\"')}"`;
  exec(`osascript -e '${script}'`, () => { /* fire-and-forget */ });
}

export function registerPollTool(server: McpServer, engine: AchievementEngine): void {
  server.tool(
    'achievement.poll',
    'Evaluate all achievement conditions against event history. Returns newly unlocked achievements (max 5 at a time). Call this after each agent turn.',
    {
      acknowledged_ids: z.array(z.string()).optional().describe('IDs of previously shown achievements to acknowledge, clearing pending queue slots for new ones'),
      limit: z.number().int().min(1).max(20).optional().default(5).describe('Maximum achievements to return'),
    },
    { readOnlyHint: true, idempotentHint: true },
    async ({ acknowledged_ids, limit }) => {
      const ack: string[] = Array.isArray(acknowledged_ids) ? acknowledged_ids as string[] : [];
      const maxResults = (limit as number) || 5;

      let pending = loadPending(engine.stateDir);

      // Remove acknowledged
      if (ack.length > 0) {
        pending = pending.filter((a: any) => !ack.includes(a.id));
      }

      // Return from pending first
      if (pending.length > 0) {
        const batch = pending.slice(0, maxResults);
        const rest = pending.slice(maxResults);
        savePending(engine.stateDir, rest);
        return {
          content: [{
            type: 'text',
            text: JSON.stringify({
              achievements: batch.map((a: any) => formatAchievement(engine, a)),
              has_more: rest.length > 0,
            }),
          }],
        };
      }

      const newlyUnlocked = engine.poll();

      if (newlyUnlocked.length === 0) {
        return {
          content: [{ type: 'text', text: JSON.stringify({ achievements: [], has_more: false }) }],
        };
      }

      // Fire macOS notification
      const cfg = loadConfig();
      const useZh = cfg.lang === 'zh';
      for (const ach of newlyUnlocked) {
        const title = useZh ? (ach.name_cn || ach.name) : ach.name;
        const desc = useZh ? (ach.description_cn || ach.description) : ach.description;
        sendMacNotification(`🏆 ${title}`, desc);
      }

      const batch = newlyUnlocked.slice(0, maxResults);
      const rest = newlyUnlocked.slice(maxResults);
      if (rest.length > 0) savePending(engine.stateDir, rest);

      return {
        content: [{
          type: 'text',
          text: JSON.stringify({
            achievements: batch.map(a => formatAchievement(engine, a)),
            has_more: rest.length > 0,
          }),
        }],
      };
    },
  );
}
```

```typescript
interface Showcase {
  slots: (string | null)[];
}

function loadShowcase(stateDir: string): Showcase {
  const p = path.join(stateDir, 'showcase.json');
  try {
    if (fs.existsSync(p)) return JSON.parse(fs.readFileSync(p, 'utf-8'));
  } catch { /* ignore */ }
  return { slots: [null, null, null, null, null, null] };
}

function saveShowcase(stateDir: string, s: Showcase): void {
  const p = path.join(stateDir, 'showcase.json');
  fs.mkdirSync(stateDir, { recursive: true });
  fs.writeFileSync(p, JSON.stringify(s, null, 2));
}

export function registerShowcaseTool(server: McpServer, engine: AchievementEngine): void {
  server.tool(
    'achievement.showcase',
    'Manage your achievement showcase (display cabinet). View current showcase, set specific slots, or auto-fill with rarest achievements.',
    {
      action: z.enum(['view', 'set', 'auto']).describe('Action to perform'),
      slot: z.number().int().min(0).max(5).optional().describe('Slot index (0–5) for "set" action'),
      achievement_id: z.string().optional().describe('Achievement ID to place in slot'),
    },
    async ({ action, slot, achievement_id }) => {
      try {
        const act = action as string;
        const showcase = loadShowcase(engine.stateDir);

        switch (act) {
          case 'view': {
            const filled = showcase.slots.map((id, i) => {
              if (!id) return { slot: i, empty: true };
              const def = engine.definitions.find(d => d.id === id);
              return def
                ? { slot: i, achievement: formatAchievement(engine, def) }
                : { slot: i, empty: true };
            });
            return { content: [{ type: 'text', text: JSON.stringify({ showcase: filled }, null, 2) }] };
          }

          case 'set': {
            if (slot === undefined || slot < 0 || slot > 5) {
              return formatMcpError(ErrorCodes.INVALID_EVENT, 'slot must be 0–5');
            }
            const aid = achievement_id as string | undefined;
            if (aid && !engine.state.unlocked[aid]) {
              return formatMcpError(ErrorCodes.NOT_FOUND, 'achievement not yet unlocked');
            }
            showcase.slots[slot] = aid || null;
            saveShowcase(engine.stateDir, showcase);
            return {
              content: [{ type: 'text', text: JSON.stringify({ status: 'ok', slot, achievement_id: aid || null }) }],
            };
          }

          case 'auto': {
            const unlocked = engine.definitions
              .filter(d => engine.state.unlocked[d.id])
              .sort((a, b) => (RARITY_RANK[b.rarity] || 0) - (RARITY_RANK[a.rarity] || 0));

            const newSlots: (string | null)[] = [null, null, null, null, null, null];
            for (let i = 0; i < Math.min(6, unlocked.length); i++) {
              newSlots[i] = unlocked[i]!.id;
            }
            showcase.slots = newSlots;
            saveShowcase(engine.stateDir, showcase);

            const filled = showcase.slots.map((id, i) => {
              if (!id) return { slot: i, empty: true };
              const def = engine.definitions.find(d => d.id === id);
              return def
                ? { slot: i, achievement: formatAchievement(engine, def) }
                : { slot: i, empty: true };
            });
            return { content: [{ type: 'text', text: JSON.stringify({ status: 'ok', showcase: filled }, null, 2) }] };
          }

          default:
            return formatMcpError(ErrorCodes.UNKNOWN_ACTION, `unknown action: ${act}`);
        }
      } catch (e) {
        return formatMcpError(ErrorCodes.STORE_ERROR, String(e));
      }
    },
  );
}
```
